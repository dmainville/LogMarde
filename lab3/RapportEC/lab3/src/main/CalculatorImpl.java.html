	public int multiply(int a, int b){
		int k=Math.abs(a);
		int p=b;
		k--;
		while(k != 0) {
           p=add(p,b);
           k--;
    }

			p=-p;	
		}
		if(b<0){
			p=-p;
		}
		return p;
	}

	public int add(int a, int b){
		int res = a;
		if (b > 0) {
			while(b-- != 0) {
	                        res++;
	                }
	        }
		else if (b < 0) {
			while(b++ != 0) {
	                        res--;
	                }
	        }
		return res;
	}
	
	public int substract(int a, int b){	
		int result = a;
		if(b > 0)
		{
			while(b-- > 0)
				result--;
		}
		else if(b < 0)
		{
			while(b++ < 0)
				result++;
		}
		
		return result;
	}
}
